# EosNsCommon-Sources library
#-------------------------------------------------------------------------------
include_directories(
  ${CMAKE_BINARY_DIR}
  ${CMAKE_SOURCE_DIR}
  ${XROOTD_INCLUDE_DIR}
  ${PROTOBUF_INCLUDE_DIRS}
  ${FOLLY_INCLUDE_DIRS}
  ${CMAKE_SOURCE_DIR}/namespace/ns_quarkdb/qclient/include
)

#-------------------------------------------------------------------------------
# Build qclient static library
#-------------------------------------------------------------------------------
add_subdirectory(ns_quarkdb/qclient)

add_library(
  EosNsCommon-Objects OBJECT

  Constants.cc
  MDException.cc
  MDException.hh

  PermissionHandler.cc                PermissionHandler.hh
  Prefetcher.cc                       Prefetcher.hh
  Resolver.cc                         Resolver.hh

  # Common helper classes, common to both namespaces
  common/QuotaNodeCore.cc             common/QuotaNodeCore.hh

  # Namespace interface
  interface/IView.hh
  interface/IFsView.hh
  interface/IFileMDSvc.hh
  interface/IContainerMDSvc.hh
  interface/Identifiers.hh
  interface/IFileMD.hh
  interface/IContainerMD.hh
  interface/IChLogContainerMDSvc.hh
  interface/IChLogFileMDSvc.hh

  # Namespace utils
  utils/DataHelper.cc
  utils/Descriptor.cc
  utils/FileListRandomPicker.cc
  utils/ThreadUtils.cc
  utils/TestHelpers.cc
  utils/Buffer.hh

  utils/Etag.cc                       utils/Etag.hh

  # non-loadable classes used in QDB namespace
  ns_quarkdb_static/accounting/ContainerAccounting.cc     ns_quarkdb_static/accounting/ContainerAccounting.hh
  ns_quarkdb_static/accounting/SyncTimeAccounting.cc      ns_quarkdb_static/accounting/SyncTimeAccounting.hh
  ns_quarkdb_static/accounting/FileSystemHandler.cc       ns_quarkdb_static/accounting/FileSystemHandler.hh
  ns_quarkdb_static/accounting/FileSystemView.cc          ns_quarkdb_static/accounting/FileSystemView.hh
                                                          ns_quarkdb_static/accounting/SetChangeList.hh

  ns_quarkdb_static/explorer/NamespaceExplorer.cc         ns_quarkdb_static/explorer/NamespaceExplorer.hh
  ns_quarkdb_static/flusher/MetadataFlusher.cc            ns_quarkdb_static/flusher/MetadataFlusher.hh

)

add_dependencies(EosNsCommon-Objects EosCliProto-Objects)

set_target_properties(
  EosNsCommon-Objects
  PROPERTIES
  POSITION_INDEPENDENT_CODE True)

add_library(EosNsCommon SHARED $<TARGET_OBJECTS:EosNsCommon-Objects> )

target_include_directories(
  EosNsCommon PUBLIC
  $<BUILD_INTERFACE:${PROJECT_SOURCE_DIR}>
  $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/utils/fmt>
  $<INSTALL_INTERFACE:include>)

target_link_libraries(
  EosNsCommon PUBLIC
  EosProtobuf
  qclient
  ${Z_LIBRARY} fmt)

set_target_properties(
  EosNsCommon
  PROPERTIES
  VERSION ${VERSION}
  SOVERSION ${VERSION_MAJOR})

install(
  TARGETS EosNsCommon
  LIBRARY DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR}
  ARCHIVE DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR})

#-------------------------------------------------------------------------------
# EosNsCommon-Static library
#-------------------------------------------------------------------------------

  add_library(EosNsCommon-Static STATIC $<TARGET_OBJECTS:EosNsCommon-Objects> )

  target_include_directories(

    $<BUILD_INTERFACE:${PROJECT_SOURCE_DIR}>
    $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/utils/fmt>
    $<INSTALL_INTERFACE:include>)

  target_link_libraries(
    EosNsCommon-Static PUBLIC
    fmt
    EosProtobuf-Static
    ${Z_LIBRARY_STATIC})

  set_target_properties(
    EosNsCommon-Static
    PROPERTIES
    POSITION_INDEPENDENT_CODE True)
endif ()
